    **Part B: Create dashboard/src/hooks/usePipelineStatus.ts**

    Follow the same pattern as useDevices/useCommands: export a hook that returns state and a handleWsEvent callback.

    ```typescript
    import { useState, useCallback } from "react";
    import type { WsEvent, PipelineState } from "../api/types";

    export function usePipelineStatus() {
      const [status, setStatus] = useState<PipelineState>("offline");

      const handleWsEvent = useCallback((event: WsEvent) => {
        if (event.type === "pipeline_status") {
          setStatus(event.status);
        }
      }, []);

      return { status, handleWsEvent };
    }
    ```

    Default state is "offline" per UI-SPEC (on mount, before any WebSocket event arrives).

    **Part C: Create dashboard/src/components/PipelineStatus.tsx**

    Create the PipelineStatus component following the UI-SPEC exactly. It renders a colored dot + label, matching the StatusBar dot+label pattern.

    ```typescript
    import type { PipelineState } from "../api/types";

    const stateConfig: Record<PipelineState, { label: string; dotClass: string }> = {
      listening:  { label: "Listening",   dotClass: "bg-emerald-500" },
      processing: { label: "Processing",  dotClass: "bg-amber-500 motion-safe:animate-pulse" },
      responding: { label: "Responding",  dotClass: "bg-blue-500" },
      error:      { label: "Error",       dotClass: "bg-red-500" },
      offline:    { label: "Offline",     dotClass: "bg-zinc-500" },
    };

    export function PipelineStatus({ status }: { status: PipelineState }) {
      const config = stateConfig[status];

      return (
        <div className="flex items-center gap-2">
          <span
            className={`inline-block h-2 w-2 rounded-full ${config.dotClass}`}
            aria-hidden="true"
          />
          <span className="hidden text-sm text-muted-foreground sm:inline">
            {config.label}
          </span>
          <span className="sr-only">{config.label}</span>
        </div>
      );
    }
    ```

    Color mapping per UI-SPEC:
    - listening: bg-emerald-500 (matches StatusBar connected dot, DeviceCard ON badge)
    - processing: bg-amber-500 motion-safe:animate-pulse (matches StatusBar connecting dot)
    - responding: bg-blue-500 (new; blue for active response)
    - error: bg-red-500 (matches StatusBar disconnected dot)
    - offline: bg-zinc-500 (matches DeviceCard OFF badge)

    No animations except pulse on processing (per UI-SPEC: "no transition animations, no queuing").
